زِنٛدگیہِ ہٕنٛدۍ پیٖچیٖدٕ واقعہٕ تہٕ کھرۍدار تَجربہٕ پیٚش کرنہٕ خٲطرٕ چھےٚ یِمَو اِشارَن تہٕ علامژَن ہٕنٛدِ ورگہٕ اِظہارٕ پَنٕنۍ تَخلیٖقی ہۄنر نیشنٲومٕژ یَتھ منٛز اَسطوٗرن ہٕنٛدِ وَلہٕ تِمو زِنٛدگیہِ ہٕنٛدۍ اَنہٲرۍ تَجربہٕ واریاہ لوٗبہٕ_وٕنہِ اَنٛدازٕ چھِ کٔرۍمٕتۍ۔ اَفسانہٕ نِگٲری پَتہٕ ترٛاوَو وۄنۍ بیٚیِس أکِس موضوٗعس پٮ۪ٹھ نَظَر یَتھ منٛز اَسہِ گَژِھ پٲرۍزان کٲشرِ شٲعری ہٕنٛز تہٕ تَتھ منٛز موجوٗد بلاغتٕچ ۔  اَکھ کِتاب ( کٲشِر شٲعری تہٕ بلاغت ) چھےٚ دۄن ہَتَن صَفَن پٮ۪ٹھ مُشتَمِل۔  زنانہٕ کٔرِن دٔج ( ڈوپٹہٕ ) دِنُک اِہتمام تہٕ اَمہِ سۭتۍ تھٲوِن پَنُن کَلہٕ تہٕ سیٖنہٕ ژھایہِ تہٕ ژوٗرِ ۔
ڈوپٹہٕ لٲگِن تِیُتھ توٚن زِ کلُک مَس یِیہِ اَتھِ۔ ڈُوپٹُک مَقصدٕے چُھ یہِ زِ اَمہِ سۭتۍ گژھہِ حُسن ژوٗرِ روزُن۔ قرانِ پاکَس منٛز چُھ ﷲ تعالیٰ فرماوان۔ تہٕ پَنہٕ_نٮ۪ن سیٖنَن پٮ۪ٹھ ترٛٲوِن پَنہٕ_نٮ۪ن ڈوپٹن ہٕنٛز لونٛچہِ “ اَکہِ دۄہ وٲژ نبیؐ نِشہِ مِصرٕ پٮ۪ٹھٕ تٔنۍ مَلمَل۔ نبی ﷺ دِیُت اَمہِ منٛزٕ کینٛہہ حِصہٕ کلبی یَس تہٕ فرمووہکھ “ اَمہِ منٛزٕ ژٔٹۍزِ کینٛہہ حِصہٕ بنٲوۍزٮ۪س   پانس کیُت کُرتہٕ۔ تہٕ اَکھ حِصہٕ دیٖزِ پَننہِ خانٛدارنہِ ڈوپٹہٕ بناونہٕ خٲطرٕ۔ اَماپوٚز تٔمِس ؤنۍ زِ اَتھ أنٛدٕرۍ بیاکھ کپُر لَگاوُن تہِ کیٛازِ جِسمٕچ أنٛدرِم تَن گژھِ نہٕ اَتھِ یِنۍ “ ( ابوٗداؤد ) کِتاب تہٕ سۄنتٕکۍ ہِدایت نَظرَن تَل تھٲوِتھ گژھن ﷲ تعالیٰ سٕنٛدۍ احکام پوٗرٕ کرٕنۍ تہٕ خۄدا تہٕ رسول خۄدا ﷺ سٕنٛدٮ۪ن احکامن گژھن نہٕ مزاق کرٕنۍ۔ حضرتِ عایشہؓ چھےٚ فرماوان زِ “ ییٚلہِ یہِ حُکم نٲزِل سَپُد۔ زنانو ترٛوو توٚن کپُر تہٕ موٚٹ کپُر ژٲرِتھ بنٲوِکھ ڈوپٹہٕ “ پلو پۄشاکھ گژھِ ہَمیٚشہِ پَننہِ طاقتہٕ تہٕ وُسعتہٕ مُطٲبِق لاگُن ۔
پٲجامہٕ یا لوٗنٛگۍ تھٲوِو ژٕننٮ۪و ہِیوٚر ۔
یِم لُکھ غوٚروٗر تہٕ تکبُرٕ خٲطرٕ پَنُن پٲجامہٕ یا لوٗنٛگۍ لاگن ژٕنٕنٮ۪و بۄن چھِ ترٛاوان ، نبی ﷺ سٕنٛزَن نَظرَن منٛز چھِ تِم لُکھ ناکام تہٕ نامُراد تہٕ سَخٕت سزابٕکۍ حقدار ۔
نبی ﷺ سُنٛد اِرشادِ پاک چُھ “ ترٛٮ۪ن قٔسمہٕ کٮ۪ن لوٗکن سۭتۍ کرِ نہٕ ﷲ تعالیٰ قیامتٕکہِ دۄہ کتھ کرِ تِمَن کُن نَظَر تہٕ کرِ تِمَن پاک تہٕ صاف کٔرِتھ جَنتَس منٛز دٲخٕل ، بٔلکہِ دِیہِ تِمَن انتہٲیی دَردناک عزاب ، حضرتِ ابوذر غفارِینؓ پرٛوژھنکھ زِ یا رسول اﷲ ﷺ ! یِم غوٚروٗر تہٕ تَکبُر سان پَنُن تِہنٛد ژٕننٮ۪و بۄن چھِ تھاوان آسان دوٚیم سُہ شَخٕص یُس پنٕنۍ احسان گٔنٛزران آسہِ ۔
( مُسلم ) ترٛیِم سُہ شَخٕص یُس اَپزی قَسم کھٮ۪تھ پٔتِس تِجارتس پرٛزلاوُن چُھ یَژھان ۔
حضرت عُبید بن خالدؓ چھِ پَنُن اَکھ واقعہ بیان فرماوان ۔ فرماوان چھِ زِ بہٕ اوسُس اَکہِ وِزِ مٔدینہٕ مٔنٛزۍ پَکان زِ مےٚ بوٗز پَننہِ پٔتۍمہِ طرفہٕ آواز “ پَنُن تِہبنٛد تھاو تھوٚد ۔ زِ اَمہِ سۭتۍ چُھ اِنسان ظٲہری نَجاستہٕ نِشہِ تہِ پاک روزان تہٕ بٲطنی نجاستہٕ نِشہِ تہِ “ بہٕ پھیوٗرُس پتھ کُن تہٕ مےٚ وُچھ نبی ﷺ ۔ مےٚ کوٚرمَکھ عرض “ یا رسول اﷲ ﷺ ! یہِ حَض چھِ اَکھ معموٗلی ژادَر ۔  اَتھ منٛز کیاہ غوٚروٗر تہٕ تَکبُر ہیٚکہِ ٲسِتھ ! “ نٔبین ﷺ فرموو اِرشاد “ کیاہ چانہِ خٲطرٕ چھِ نا میٲنۍ پٲروی کرٕنۍ ضوٚروٗری۔ مےٚ بوٗزۍ نبی ﷺ سٕنٛدۍ اَلفاظ تہٕ مےٚ پیٚیہِ تِہنٛدِس بَنٛدس کُن نَظَر مےٚ وُچھ تِہُنٛد ﷺ تہبنٛد نٮ۪صف گرٛۄزِ ہیوٚر۔اَمہِ کٔمی باوجوٗد چُھ یہِ باب کٲشرِس نثرَس کیوٚو کٲشرِس تَنقیٖدَس اَکھ اَہَم اِضافہٕ یُس توٚتام پَنٕنۍ اِفٲدیت بَرقرار تھاوِ یوٚتامت دُنیاہَس منٛز شعر وَنُن باقٕے روزِ۔ زیٚر بَحث کِتابہٕ ہٕنٛز سارِوٕے کھۄتہٕ بٔڑ لگہٲری چھےٚ یہِ زِ یہِ کرِنوٮ۪ن شٲعرن تہٕ اَدب_نوازَن شعرَس تہٕ اَدبَس مُتعلِق مشرقی تَصوُرن ہٕنٛز زان تہٕ تِمَن دِیہِ ییٚمہِ کتھِ ہٕنٛز وٲقفِیَت زِ شعر مَحض خیال چہِ خوٗبصوٗرتی یا زبٲنۍ ہٕنٛدِ سطحی خوٗبصوٗرتی سۭتۍ بارسَس یِوان بٔلکہِ چھےٚ شعرٕچ زَبان عموٗمی زبٲنۍ وَرتاوٕ مقابلہٕ اَکہِ زوردار تہٕ طاقتور ورتاوٕچ زَبان آسان۔شٲعر چُھ خیالَس مقابلہٕ کۭفیٔژ ہٕنٛدِس وۄتلنَس تَرجی دِوان تہٕ شعرَس منٛز چھےٚ کۭفیَت تیٚلہِ پٲدٕ ہٮ۪کان سَپدِتھ ییٚلہِ تَتھ منٛز ورتاونہٕ آمٕژ زَبان سٮ۪ٹھہٕے منٲسِب تہٕ تَھدِ وَرتاوٕچ آسہِ ۔
کِتابہِ منٛز مُختٔلِف مسلَن ہٕنٛز ویژھنَے کران چھِ مُصنِفَن بعضے تِتھۍ شعر تہِ پیٚش کٔرۍمٕتۍ یِمَن ہٕنٛدِس شعرتِس پٮ۪ٹھ پرَن وٲلِس بعضے شَک گژھان چُھ ۔
تاہَم یۄسہٕ محنت تہٕ مُطالعہٕ مُصنِفَس کِتاب لٮ۪کھنہٕ خٲطرٕ کرُن پیوٚمُت چُھ سُہ چُھ ییٚمہِ کتھِ ہُنٛد ٲنہٕ زِ مُصنِفَن چُھ پَننہِ بِترِ تہٕ مقدوٗرٕ مُطٲبِق پَننہِ مطالعُک حٲصل کٲشرِ زبٲنۍ منٛز پھیُرمُت سُہ تہِ تَمہِ وِزِ ییٚلہِ تِمَن اَتھ سِلسِلَس منٛز کانٛہہ مُستحکم رٮ۪وایت ڈکھِ ٲس ۔
کٲشرِس منٛز چُھ شعر مۄلوَنہٕ کہِ حوالہٕ واریاہ مواد موجوٗد ، مگر بلاغت تہٕ اَمہِ کٮ۪ن مُختٔلِف سرچٔشمن ہٕنٛدِ حوالہٕ اوس نہٕ زیٚر بَحث کِتابہِ برٛۄنٛہہ کانٛہہ خاص تہٕ نثری سرمایہِ موجوٗد ماسواے راہی صٲبٕنۍ کِتاب یَتھ پٮ۪ٹھ برٛۄنٛہہ پَہن کتھ آیہِ کرنہٕ ۔
بَہرحال مےٚ چھےٚ وۄمید زِ نٲظر صٲبٕنۍ کِتاب “ کٲشر شٲعری تہٕ بلاغت “ سَپدِ کٲشرٮ۪ن اَدبی حلقَن منٛز مقبوٗل تہٕ اتھ میٚلہِ پزیرٲیی تہٕ کٲشرٮ۪ن طٲلب علمَن ہٕنٛز اَکھ نصٲبی ضروٗرت تہِ کرِ یہِ کِتاب پوٗرٕ۔نوٚو اِنقلاب زراعتی اِنقلاب تہٕ صنعتی اِنقلابہٕ پَتہٕ یُس اِنقلاب دُنیاہَن پٔتۍمِس دٔہۍلِس منٛز محسوٗس کوٚر ، سُہ چُھ دَراَصل انفارمیشن نٹیکنالوجی سۭتۍ رونما سَپدَن وول یۄہَے معلوٗمٲتی اِنقلاب۔یہِ معلوٗمٲتی اِنقلاب سَپُد تَتھ تیزرفتار تہٕ غٲر محسوٗس طٔریٖقَس پٮ۪ٹھ بَرپا کہِ اِنسٲنی عقٕل گٔیہِ حٲران۔
اَمہِ اِنقلاب کِس نٔتیٖجَس منٛز سَپُد یَتھ بُترٲژ پٮ۪ٹھ تیُتھ اَکھ عٔلمی دماکہٕ رونم ییٚمۍ زَن اِنسٲنی زِنٛدٕگی ہٕنٛدِس پرٛٮ۪تھ گوشش اَکھ نوٚو رۄخ دِیُت۔
نٔبین ﷺ ہُنٛد چُھ اِرشاد “ ژٕننٮ۪و ہیوٚر پٲجامہٕ یا لوٗنٛگۍ تھاونہٕ سۭتۍ چُھ اِنسان پرٛٮ۪تھ ظٲہری تہٕ بٲطنی نَجاستہٕ نِشہِ محفوٗظ روزان “ مَطلَب چُھ یہِ زِ ییٚلہِ پٲجامہٕ یا لوٗنٛگۍ ژٕننٮ۪و بۄن روزِ یہِ گژھِ وَتہِ پکنہٕ سۭتۍ مٔکٕر تہٕ پاک تہٕ صاف روزِ نہٕ تہٕ یہِ کتھ چھِ طہارتہٕ کٮ۪ن اوٚصولَن خٕلاف ۔
مومِن سٕنٛدِ خٲطرٕ بِہتٔریٖن نوٚموٗنہٕ “۔ ( القرآن ) نبی ﷺ ہُنٛد اِرشاد چُھ زِ “ مومِن سُنٛد تہبنٛد گژھہِ نٮ۪صف گرۄزِ تام آسُن تہٕ ژٕننٮ۪ن تام آسہٕ نَس تہِ پرواے ۔
اَماپوِز ژٕننٮ۪و بۄن یوٗتاہ تِہبنٛدس حِصہٕ روزِ سُہ دَزِ نارس منٛز تہٕ یُس شَخٕص غوٚروٗرٕ تہٕ تَکبُرسان پَنُن تہبنٛد ژٕننٮ۪و بۄن آسہِ ترٛاوان ، قیامتٕکہِ دۄہ وُچھِ نہٕ اﷲ تعالیٰ تٔمِس کُن “ ( ابوٗداوٗد ) پٲٹۍ پَلو لٲگِو یہِ چُھ زَنانَن ہُنٛد پۄشاکھ تہٕ نٔبین ﷺ چُھ مردَن زَنانَن ہُنٛد پۄشاکھ تہٕ شَکٕل و صوٗرت بناونَس سختی سان منع کوٚرمُت۔
حضرت عمرؓ چھِ فرماوان زِ نٔبین ﷺ ہُنٛد اِرشاد چُھ “ پٲٹۍ پلو لٲگِو زِ یُس شَخٕص اَتھ دُنۍیاہَس منٛز لاگہِ سُہ ہیٚکہِ نہٕ یہِ ٲخرتَس منٛز لٲگِتھ “ ( بخاریمسلم ) اَکہِ دۄہ فرموو نٔبین ﷺ حضرت علیؓ یَس : یہِ پٲٹۍ کپُر ژَٹھ اَتھ کَلہٕ ژادٕر ( ڈوٗپتہٕ ) بنٲوِتھ کر یہِ یِمَن خوتون منٛز تَقسیٖم “ ( مُسلم ) اَمہِ سۭتۍ سَپُد یہِ معلوٗم زِ زَنانَن خٲطرٕ چُھ پٲٹۍ کپُر لاگُن ٹھیٖک ۔
توے دِیتُوٕ تۄہہِؐ حُکم زِ خوٚتوٗنَن خٲطرٕ بنٲیوکھ کَلہٕ ژادرٕ ۔
اَمہِ ورٲی یی ہَے نہٕ یہِ دوٚیمٮ۪ن کامٮ۪ن تہِ اِستعمال کرنہٕ ۔
زنانہٕ لٲگِن توٚن کپُر ییٚمہِ مٔنٛزۍ جٕسمٕچ بناوَٹھ ظٲہر سَپدِ ۔
نبی ﷺ چھِ اَمہِ قٔسمہٕ چٮ۪ن زنانَن عبرتناک نٔتیٖجٕچ ژینٕؤنۍ دِژمٕژ ۔
اَکہِ دۄہ آیہِ حضرت اسماؓ توٚن پوشاکھ لٲگِتھ حضرت نبیؐ سٕنٛدِس خٔدمتَس منٛز ۔
تٔمۍ کٔر پرُس آسٹریلِیاہَس خٕلاف شارجاہَس منٛز لگاتار ترٛے سینٛچرییہِ ییٚمہِ سۭتۍ سُہ دُنۍیِہُک اَکھ بوٚڑ کِرکٹ کھِلٲڑۍ ماننہٕ آو ۔
سٔچِن سٕنٛسز خصوٗصِیَت چھِ یہِ زِ سُہ چُھ پرٛٮ۪تھ کٲنٛسہِ بالرس سۭتۍ گِنٛدنَس منٛز سٮ۪ٹھاہ مٲہِر تہٕ تٔمۍ چُھ دُنۍیاہ کٮ۪ن سرکردٕ سِپنَرن پَننہِ اعلی مہارتُک دابہٕ بہنومُت ۔
اَوٕکِنۍ چھِ اَکھ حٔقیٖقت زِ سٔچِن ٹنٛڈولکر چُھ ییٚمہِ زمانُک اَکھ بےمِثال کرکٹ کھِلٲڑۍ ۔
وۄنۍ کرَو أسۍ اَمہِ پَتہٕ کتھ مُکہٕ بٲزی ہٕنٛدِس بارَس منٛز یَتھ منٛز اَگَر غٲرزانَس تِھی تہِ ہیمَو ٹایسَن سُنٛد ناو سُہ تہِ آسہِ زانان زِ ٹایسَن کِیُتھ پرٛزلوُن تارُک اوس مُکہٕ بٲزی منٛز ۔
مَگَر بَدقٔسمٔتی سَببہٕ گٔیہِ تٔمِس سۭتۍ تہِ تِتھۍ حالات ییٚمہِ سۭتۍ تٔمۍ سُنٛد ناو تہِ گوٚو ہَنا بَدنام پَہَن ۔
مگَر أسۍ کیاہ ہٮ۪کو ؤنِتھ کیازِکہِ میٖڑِیاہَس گَژِھ صِرف اَکھ کتھ اَتھِ لَگٕنۍ تہٕ تِم کرٕہَن تَتھ کُنہِ تہِ طٔریٖقہٕ دُنۍیاہَس برٛۄنٛہہ کُنہِ تِتھ پٲٹھۍ پیٚش یَتھ منٛز تِم کرٕہَن پرٛٮ۪تھ طٔریٖقہٕ پَنُن نوٗن مَرژٕوانٛگُن رَلٲوِت سارِنٕے برٛۄنٛہہ کُنہِ پیٚش ۔
یِژھٕے ہِش بٲش س گٔیہِ ٹایسَنَس سۭتۍ تہِ ۔
ٹایسن مُکہٕ بٲزی ہُنٛد عالمی چمپین محمد علی کِلے یَس پَتہٕ ییٚمۍ سیاہ فارمولہٕ ویٖرَن سٲرسٕے دُنۍیاہَس منٛز پَننہِ جِسمٲنی طاقتہٕ تہٕ فولٲدی اَتَھو سۭتۍ مُکہٕ_بٲزی ہٕنٛدٮ۪ن عاشقن ہٕنٛدۍ دِل زیٖنۍ سُہ چُھ مایک ٹالیسن یعنی مسلمان بَننہٕ پَتہٕ عبدال عزیٖز ٹایسن سٕنٛدِ ناوٕ پرٛزلان ۔
ٹایسن چُھ ذٲتی طور تہِ محمد علی کِلے سُنٛد ظٲہری تہٕ بٲطنی عظمتہٕ سۭتۍ صِرف مُتٲثِر بٔلۍکہِ تَمہِ سۭتۍ اَکھ زبردَس روحٲنی تہٕ جِسمٲنی قوت حٲصِل کران ۔
دۄشوٕنۍ پایہِ بَڈٮ۪ن چھِ اَکھ کتھ مُشترکہٕ یعنی محمد علی کِلے یَس روٗدۍ سفید ژمہِ ہٕنٛدۍ مغرِبی اِجارٕدار ہمیٚشہٕ مُخٲلِف تہٕ رنٛگہٕ رنٛگہٕ اوٚنُکھ تنٛگ ۔
سُہ چُھ نامور عالمی کرکٹ چمپین سر رونالڈ برٮ۪ڈ مین ہِوِس عظیٖم بیٹس مینَس ہَم_پَلہٕ ماننہٕ یِوان ییٚمۍ سُنٛد ریکارڈ ۹۴ ـ ۹۹ چُھ ۔
سچن ٹیِنٛڑولکر آو کِرکٹ کِس مٲدانَس منٛز ۱۹۸۹ منٛز تہٕ پٔتۍمٮ۪ن کَہَن ؤرۍ یَن دِتۍ تٔمۍ پننہِ گِنٛدنٕکہِ اَنٛدازٕ سۭتۍ ثٲبِت کٔرِتھ زِ سُہ چُھ اَکھ تِیُتھ بیٹَس مین یُس پٔزۍ پٲٹھۍ عظیم وننَس لایَق چُھ ۔
اَکہِ دۄہ کِس بینالا قوامی کرکٹ میٚچس منٛز چُھ سُہ سارِوٕے کھۄتہٕ جادٕ سنچری بناون وول اَکھ ناقٲبِل تسخیٖر کھِلٲڑۍ ۔
یِتَھےکٔنۍ چُھ سٔچِن اَکہِ دۄہ کِس بینالا قوامی کِرکٹ میٚچس منٛز سارِوٕے کھۄتہٕ جادٕ رَنہٕ بناونَس منٛز عالمی سوٚتھرَس دویٚمِس دَرجَس پٮ۪ٹھ ۔
سٔچِن سٕنٛز بیاکھ خصوٗصِیَت چھےٚ یہِ زِ سُہ چُھ عالمی سوٚتھرِس پٮ۪ٹھ سارِوٕے کھۄتہٕ جادٕ لَٹہِ مین آف دی میٚچ قرار دِنہٕ آمُت ۔
سٔچِن ٹیِنٛڈولکرَن چُھ ٹٮ۪سٹ کِرکٹس منٛز ۵۸۹۰ رَنہ ۲۲ سینٛچُری بنٲوِتھ ۲۳ / ۵۰ کہِ مددتہٕ ۱۸ ۔
۵۶ اوسط سۭتۍ اَکھ عالمی ریکارڈ قٲیِم کوٚرمُت ۔
سٔچِن تیِنٛڈولکرَن چُھ ہِنٛدُستٲنۍ کرکٹ کِس توٲریٖخَس منٛز سارِوٕے کھۄتہٕ جادٕ سِکور بنومُت ۔
یُس ۱۸۶ ناٹ آوٹ چُھ ۔
سُہ چُھ کامیٲبین ہٕنٛدِس اعلی معیارَس تام واتنہٕ خٲطرٕ بَرابَر جِدوجہد کران ۔
سارِنٕے کھیلَن چھےٚ پَنٕنۍ اَکھ اَلَگ اٮہمِیَت تہٕ ضوٚروٗرَت ۔
سٲری لُکھ چھِ یِم کھیلہٕ بَڑٕ شوقہٕ تہٕ زوقہٕ سان گِنٛدان تہِ وُچھان تہِ ۔
لُکھٕے یٲتۍ کینٛہہ بٔلۍکہِ مُختٔلِف مُلک تہِ چھِ مُختٔلِف کھیلَن منٛز پانہٕ وٲنۍ بَڑٕ زورٕ شورٕ سان حِصہٕ نِوان ۔
کُنہِ وَقتہٕ چھِ سِری_لَنکا تہٕ پاکستان پانہٕ وٲنۍ میٚچ گِنٛدان تہٕ کُنہِ وَقتہٕ اِنٛڑیا تہٕ سَوُتھ افریٖکا ۔
یِم کھیلہٕ چھےٚ یِوان ٹیوی یَس پٮ۪ٹھ تِہ ٹیٚلِکاسٹ کرنہٕ ۔
یِتھ پٲٹھۍ چھِ پوٗرٕ دُنۍیِہٕکۍ لُکھ یِمَن کھیلَن ہُنٛد مَزٕ تہٕ لُطُف تُلان ۔
وۄنۍ اَگَر وُچَھو مُختٔلِف چیٖزَو تہٕ تَنازَو چُھ اَسہِ پانہٕ وٲنۍ تیوٗتاہ دوٗرٮ۪ر کوٚرمُت ییٚمۍ کِس نٔتیٖجَس منٛز اَسہِ منٛز وۄنۍ تیوٗت مِلہٕ ژار پانہٕ وٲنۍ مُلکن دَرمیان روٗدمُت ۔
تِمَن ہٕنٛز مِثال کرَو أسۍ یِتھ پٲٹھۍ بَیان : آسٹریلیا چہِ ہاکی ٹیٖمن کوٚر بھارت یِنہٕ نَشہِ اَنٛکار سڑنی ، آسٹریلیا ہَن چھےٚ بھارتہٕ کِس ژور مُلکی ٹورنامنٹَس منٛز گِنٛدنہٕ نِش معزرت ظٲہِر کٔرمٕژ ۔
چٔسندیگَڑھَس منٛز نَوِ رٮ۪تہٕ پٮ۪ٹھ مُنعقد سَپدَن وٲلِس ٹورنامنٹس منٛز ٲس پٲکستانٕچ شٔرکت مشکوٗک گٔمٕژ ۔
آسٹریلیا ہٕکۍ ہاکی فیڑریشنَن چُھ چٔندیگَڑھ کِس ٹورنامنٹس منٛز پٲکستان چہِ جایہِ گِنٛدنَس اِنکار کوٚرمُت تِکیازِ ممبَی واقَو چُھ تِمَن بھارتس منٛز حفاظتی خدشہِ پٲدٕ کوٚرمُت ۔
صابنہِ وول چُھ آسان صابَن ہٮ۪تھ تہٕ تولِیاہ ہٮ۪تھ ۔
تَھالہٕ وٲلۍ چھِ تھال ترٛاوان تہٕ اَمہِ پَتہٕ چھےٚ سِنۍ والٮ۪ن ہٕنٛز رٔج لَگان ۔
سِنۍ والٮ۪ن پَتہٕ چھِ یِوان بَتہٕ ترٛاوَن وٲلۍ ۔
تہٕ تِمَو پَتہٕ وازٕ قٔلۍیہِ دیگ ہٮ۪تھ یِوان تہٕ قٔلۍیہِ ترٛاوان ۔
بَتہٕ کھیوٚن چُھ شروٗع گژھان ۔
مَہرازٕ اَمہِ ساتہٕ بَتہٕ کھٮ۪وان ۔
کھیل کوٗد اِنسان سُنٛد یَتھ دُنۍیا ہَس منٛز روزُن ، ییٚتہِ وۄتُھن بِہُن تہٕ باقٕے کامہِ کار کرنہٕ علاوٕ چُھ أکِس اِنسانَس پَننہِ صحتُک تہِ خیال تھاوُن آسان ۔
زِنٛدگی صحت مَنٛد تہٕ صحت یاب تھاونہٕ خٲطرٕ چُھ أکِس اِنسانَس تِم چیٖز ضوٚروٗری آسان کرٕنۍ یِم تٔمۍ سٕنٛدِس صحتَس کانٛہہ تہِ زَرٮ۪ر واتناوَن نہٕ کینٛہہ بٔلۍکہِ سَپدٮ۪س ہُرٮ۪ر صحتَس منٛز ۔
یِمنٕے چیٖزَن منٛز چُھ یِوان اِنسان سُنٛد گِنٛدُن تہٕ درٛۄکُن بیترِ ۔
یِتھٕ پٲٹھۍ وُچَھو ، اَسہِ برٛۄنٛہہ کُنہِ چھِ مُختٔلِف قٔسمَن ہٕنٛدۍ مُختٔلِف کھیلہٕ یِمَن منٛز کرکٹ میٚچ ، ہاکی ، فُٹبال بیڑمِنٹَن ، ٹیبٕل ٹینٕس بیترِ چھِ یِوان ۔
یہِ کتھ چھےٚ ژٮ۪تس پاوٕنۍ لایَق زِ چٔندی_گڑھ کِس اَتھ ٹورنامنٹس منٛز چُھ ہالینڑَس تٕہ جرمنی تہِ حِصہٕ نیُن ۔
پٲکستٲنۍ آی پی ایل کھِلٲڑۍ ذہنۍطور بھارتس منٛز گِنٛدنہٕ باپَتھ ناتیار لاہوٗر ، پٲکستٲنۍ انڑین پریمر لیگ ( آی پی ایل ) کھلاڑٮ۪و چُھ یِنہٕ واجنہِ آی پی ایل لِیٖگہِ منٛز شٔرکت کرنہٕ باپَتھ پَنُن ذہَن تیار کوٚرمُت ۔
تِہُنٛد وَنُن چُھ زِ پٲکستانَس تٕہ بھارتس دَرمیان مُژُکری گَژھنہٕ پَتہٕ تِہُنٛد بھارتَس منٛز گِنٛدنُک سوالٕے پٲدٕ گَژھان ۔
یہِ کتھ چھےٚ ثٮ۪قہٕ زِ آی پی ایلُک دوٚیِم ایڑیشن شروٗع گَژھنَس چھِ وُنہِ ترٛے رٮ۪تھ تہٕ مُمکِن چُھ زِ توٚتام ما گژھن حالات بیٚیہِ ٹھیٖک وۄں_گوٚو تیٚلہِ تہِ آسہِ حکوٗمت طرفہٕ اِجازت دَرکار ۔
کپتان شعیب ملک سُنٛد وَنُن چُھ زِ آی پی ایل چُھ اَکھ دماکہٕ خیز ٹورنامنٹ یُس کارکردگی تہٕ مٲلۍ لحازا کھلاڑین خٲطرٕ واریاہ سوٗدمنٛد چُھ وں_گوٚو وۄنۍ چھِ حالات بدلیٚمتۍ ۔
بھارتی ٹیٖمن کوٚر پٲکستان یِنَس انٛکار ییٚمہِ پَتہٕ دۄشوٕنۍ مُلکَن ہٕنٛدٮ۪ن کِرکٹ تَعلقاتَن منٛز مُژُکری پٲدٕ گٔیہِ تہٕ پٲکستٲنۍ کھلاڑین ہُنٛد بھارتَس منٛز گِنٛدُن چُھ مُشکِل ۔
وۄنۍ ترٛاوَو أسۍ اَکھ نَظَر ٹینٕس کِس دُنۍیا ہَس کُن تہِ یَتھ منٛز حیدر آباد چہِ سانِیا مِرضا ہَن پَنُن ناو کھول پٔزۍ پٲٹھۍ آسمانَس پٮ۪ٹھ ۔
مَگَر بَدقٔسمٔتی سَببہٕ گٔیہِ سانِیا مِرضا زَخمی تہٕ ہیوٚکُکھ نہٕ تٔمۍ پَتہٕ کینٛہَن ٹورنامینٹَن منٛز حِصہٕ نِتھ ۔
ثانیہ مِرزا آسٹریلین اوپن ٹورنامنٹ خٲطرٕ مدعو حیدرآباد ، زخمی آسنٕہ کِنۍ ژۄن رٮ۪تن ہٕنٛدِس وَقفَس دوران ٹینس کہِ عالمہٕ نِش دوٗر ثانیہ مرزا یہِ وٲژ تَمہِ ساتہٕ راحت ییٚلہِ تَس پٔتۍمِہ ہَفتہٕ یہِ اِطلاع دِنہٕ آیہِ زِ نَوِ رٮ۪تہٕ میلبورنَس منٛز یِیہِ تٔمِس سیٚدِ سیوٚد دٲخلہٕ دِنہٕ ۔
ذرایع ابلاضس سۭتۍ یِمَن کتَھن ہٕنٛز باوَتھ کران کوٚر ثانیہ مرزایہِ اَتھ خَبرِ پٮ۪ٹھ مسرتُک اِظہار ۔
چایہِ وٲلۍ چھِ چاے ہٮ۪تھ یِوان یہِ چھےٚ لِپٹَن چاے آسان ۔
مُشکہٕ تُجہِ چھےٚ آسان مُشِکۍ ژھٹان ۔
چایہِ وٲلِس پَتہٕ پَتہٕ چُھ آسان بِسکوٗٹہٕ وول ۔
سالَر چھِ منشا موٗجوٗب بِسکوٗٹ جوراہ جوراہ تُلان ۔
تَمہِ پَتہٕ چُھ بیاکھا یِوان دال موٗنگرا ہٮ۪تھ ۔
سٲری چھِ مۄٹھا مۄٹھا تُلان ۔
تَمہِ پَتہٕ چُھ مِٹھایہِ وولاہ پَکان ۔
سٲری چھِ منشا موٗجوٗب رَژھا رَژھا تُلان ۔
مَہرازٕ یِمَو چیٖزَو منٛزٕ کینٛہہ تہِ کھٮ۪وان ۔
تَس چُھ آسان فاقہٕ ۔
ٹایسنَن چِھِ اَزتام ۴۷ مقابلہٕ زیٖنۍمٕتۍ تہٕ صِرف ترٛٮ۪ن مقابلَن منٛز ہوٗرمُت سُہ تہِ نسلی بِناہن پٮ۪ٹھ ریفریٖیَن ہٕنٛز جٲنِب_دٲری کینٛہِہ کتھ چَھ وَنٕنۍ لایَق زِ برطانوی مُکہٕ باز جولیس فرانٛسس سٕنٛدِ بوحٲمی نَسل پرستَو اوس ٹایسن سُنٛد بُت جانور سٕنٛدۍ پٲٹھۍ پنٛجرَس منٛز بَنٛد کوٚرمُت یُس ٹایسنَن پَننہِ آشنہِ سۭتۍ پانہٕ تہِ وُچھ تہٕ ژَکھ کَھسنہٕ ورٲے گوٚو فرانٛسس ہارناونہٕ خٲطرٕ جادٕ مضبوٗط ۔
دُنۍیاہَس منٛز چھِ مُختٔلِف عجوٗبہٕ مُختٔلِف جایَن منٛز موٗجوٗد ۔
تِم چیٖز یِم پانَس منٛز اَکھ مِثال پٲدٕ کرَن ، چھِ پرٛٮ۪تھ وِزِ سارِنٕے لوٗکن خٲطرٕ اَکھ عجوٗبَے آسان تہٕ یِم چیٖز چھِ پرٛٮ۪تھ کانٛہہ اِنسان بَڑٕ زوقہٕ تہٕ شوقہٕ سان وُچھنٕچ خٲہِش تھاوان ۔
یِمَن چیٖزَن نَہ چُھ کانٛہہ جورٕ آسان تہٕ کانٛہہ مُشابَت ۔
یِمَن چیٖزَن منٛز چھِ یِوان دُنۍیِہٕکۍ تِم سَتھ عجوٗبہٕ یِمَن وٕچھنٕچ تَلخی چھِ پرٛٮ۪تھ أکِس اِنسانَس آسان ۔
یِمَن منٛز چُھ یِوان سانہِ ہِنٛدوستانُک تاج محل ( آگرا ) ، اہرام ( مِصٕر ) ، ہینگِنٛگ گارڑٕنٛز ( بیبیلان ) سٹیچوٗ آف زِیَس , ( اولَمپِیا ) , ٹیمپٕلآفآرمِٹٕس ، پھیروس آف ایلیکزنٛڑرٛیا ، کولوسَس ( روڑُس ) ۔
یِمنٕے چیٖزَن منٛز کرَو أسۍ سارِنٕے برٛۄنٛہہ مِصرٕکین اہرامَن ہٕنٛز کتھ : اہرام ۔
اَکھ عجوٗبہٕ دُنۍیِہکٮ۪ن عجٲٮ۪باتَن منٛز چھِ مِصرٕکۍ اہرام / مقبرٕ تہٕ مندر / تہِ یِوان ۔
یِم کر بنٲوۍ تہٕ تِم کم ٲسۍ , اَتھ پٮ۪ٹھ چُھ وارِیاہ کیِنٛہہ لٮ۪کھنہٕ آمُت ۔
کینٛہہ تَحقیٖق کرن وٲلۍ تہٕ ژھونٛڈٕ ژھانٛڈ ہٕنٛدۍ شوقیٖن چھِ وُنہِ تہِ ؤنۍ دِوان ۔
ٹایسن سٕنٛز زِنٛدگی چھِ امی نسلی تعصبہٕ سۭتۍ المناک بناونہٕ آمٕژ ۔
کیِنٛہہ کال برٛونٛٹھ آو سفید فام لابی ہٕنٛدِ طرفہٕ ٹایسن اَکھ اَپُز ریپ کیس کرنہٕ تہٕ سُہ آو جیل بَرنہٕ ۔
تَمہِ پَتہٕ کوٚرُکھ أکِس معموٗلی ٹریفِک معاملَس منٛز گرفتار تہٕ تھووُکھ ؤرۍیَس جیلَس منٛز ۔
ٹایسن آو وارِیاہَس کالَس ہالی فیلڈ ناوٕ کِس مَدِ مُقابلہٕ مُکہٕ بازَس کنَس ژوٚپ ہٮ۪نہٕ کِس اِلزامَس پٮ۪ٹھ مُکہٕ بٲزی نِشہِ معطَل کرنہٕ ۔
یِمَو سارِوٕے حٲرۍثاتَو یوِدوے ٹایسن تَنٛگ أنِتھ کُنہِ وَقتَس بےِاعتدال تہِ بنوو تہٕ تَس ٲسۍ معموٗلی کتَھن پٮ۪ٹھ ژَکھ کَھسان اَماپوٚز پَنہٕ نٮ۪ن مُداحن تہٕ قدردانَن ہٕنٛدِ اِخلٲقی حِمایتہٕ سۭتۍ گو سُہ بیٚیہِ تَروتازٕ تہٕ تٔمۍ کوٚر جنؤری ۲۰۰۰ منٛز برطانیہِ کِس قویہِ_یکل جولیس فرانَسس ناوٕ کِس مُکہٕ_بازَس سۭتۍ مُقابلہٕ کرنُک چیلینٛج قبول تہٕ سۭتۍ ووٚنُن یہِ تہِ زِ شاید پیٚیہِ اَتھ مُقابلَس منٛز برطانیہ کِس قومی باکسر جولیس فرانٛسس زِنٛدگی نِشہِ اَتھٕ چَھلُن ۔
ٹایسن سٕنٛزِ اَتھ کتھِ پٮ۪ٹھ گٔیہِ سٲرٕے مغرِبی لابی بیٚیہِ تَس مُخٲلِف تہٕ رنٛگہٕ رنٛگہٕ پرٛوپگنٛڈاہ آو تَس خٕلاف کرنہٕ ۔
گۄڈٕ آیوو یہِ مقابلہٕ یہِ ؤنِتھ مُلتوی کرنُک فٲصلہٕ کرنہٕ اَماپوِز اَمہِ مقابلہٕ کیوو مُنتَظِمَو آسہٕ سارے تیٲرِیہِ کرِمَژٕ تہٕ ۲۱۰۰۰ ٹِکٹہٕ تہِ کٕنۍمَژٕ اَوٕکِنۍ پٮ۪و تِمَن یہِ مقابلہٕ اعلانہٕ مطٲبِق کرناوُن یِتھ کٔنۍ آو ٹایسن تہٕ جولیس فرانٛسس ۲۹ جنؤری ۲۰۰۰ بَٹٕوارِ دۄہہٕ مانچسٹر شَہرَس منٛز اَکوُہہ ساس / ۲۱۰۰۰ / ۲۱... ساس حٲضِریٖنَن ہٕنٛزِ موٗجوٗدگی منٛز رنٛگہٕ منٛز مقابلہٕ خٲطرٕ ۔
تماشہٕ بیٖنَن منٛز ٲسۍ ہالیٖ_وُڈٕکۍ واریاہ فلم_سٹار , , نامور فُٹبال پلیر ڈیوڈمین تہٕ تٔمۍ سٕنٛز زنانہٕ مشہوٗر سپاٹیس گرل وِکٹوٚریہِ تہِ شٲمِل ۔
جولیس فرانٛسَس ژاو رنٛگہٕ منٛز برطانوی فوجٕکۍ کینٛہہ کمانٛڈوز باڈیٖگاڈ ہٮ۪تھ اَمہِ ورٲے آو بَرطانیہ کہِ طرفہٕ ہوٲیی حملُک سایرن واینہٕ تہٕ دَہشت بٮ۪ہہٕ ناونہٕ ٹایسن اوس معموٗلہٕ کھۄتہٕ واریاہ نارمَل باسان مقابلہٕ شروٗع سپدنٕکٮ۪ن گۄڈٕ_نہِ_کٮ۪ن ۳۰ہہن سیکنٛڈن منٛزٕے آو جولِس فرانٛسس ہِیُو دیوقامَت باکسر ٹایسن سٕنٛدِ گۄڈٕنکی دۄگہِ سۭتۍ پَتھر لاینہٕ ژورَن مِنٹَن ہٕنٛدٮ۪ن دۄن راونٛڈَن منٛز پیوٚو جولیس فرانٛسس کُنہِ خاص مزاحمتہٕ ورٲے ٹایسن سٕنٛدٮ۪و فولٲدۍ مُکو سۭتۍ سَتہِ فِرِ پَتھر تہٕ باسان اوس زِ سُہ چُھ ٹایسَن پادَن پٮ۪ٹھ زُوُک اَمن منٛگان ۔
یِتھٕ کٔنۍ آو عبدال عزیٖز ٹایسن کُنہِ تنازعہٕ ورٲے عالمی سوٚتھرِس پٮ۪ٹھ مُکہٕ بٲزی ہُنٛد رُستُم زَمان یعنی ہیویویٹ چمپین ماننہٕ پوٚز بےِتاج بادشاہ سٕنٛدۍ پٲٹھۍ درٛاو_رنٛگہٕ منٛز جزبٲتی گژھنہٕ ورٲے یہِ وَنان وَنان زِ عالمی اعزازُک تاج واپَس حٲصِل کرنہٕ مۄکھٕ روزِ میٲنۍ جدوجہد جٲری ۔
تٔمۍ سٕنٛدِ خٲطرٕ چھِ ٹرٛے منٛزٕ بَرفی پلیٹہٕ جوراہ واتان ۔
سۭتی چُھس رۄپہٕ سٕنٛدِس گِلاسَس کٮ۪تھ دۄد پیٚش یِوان کرنہٕ ۔
یہِ چُھ یہِ دۄدٕ ہَن چٮ۪وان ۔
مِٹھایہِ رَژھا کھٮ۪وان ۔
یہِ مِٹھٲۍ کھٮ۪نَس منٛز چھِ تٔمِس یار دوس تہِ مَدد کران ۔
مگر تیوٗت کٔرِتھ تہِ چُھ یِوان دۄشوٕنۍ پلیٹَن منٛز أڑۍ مِٹھٲۍ أتۍ تھاونہٕ ۔
یۄسہٕ پَتہٕ مَہریٚنہِ چھےٚ یِوان دِنہٕ ۔
اگَر نہٕ تہِ کرِ تیٚلہِ وَننَس زِ مَہرازٕ چُھ بۄچھ ۔
کپ پَلیٹہٕ چھِ یِوان ہَٹاونہٕ تہٕ چھِ اَتھ سۄرشاونہٕ خٲطرٕ نَفَر جوراہ اَژان ، یِمَن چھےٚ أکِس آسان اَتَھس منٛز چِلَمچی تہٕ بیٚیِس چُھ آسان آبہٕ جَگ ۔
مَہرازٕ سٕنٛدِ پٮ۪ٹَھے چُھ سارِنٕے یِوان اَتَھن آب ترٛاونہٕ ۔
ثانیہ مرزایہِ ووٚن یہِ تہِ زِ وۄنۍ چھےٚ سۄ صحت_یاب تٔمۍ ووٚن یِتہِ زِ یوٗ ایس اے یِچ وانیہ کنٛگ آسہِ ڈبلز ایونٹَس منٛز تٔمۍ سٕنٛز حِصہٕ_دار ییٚمِس سۭتۍ تٔمۍ مورو کواوپنڈ بلٹایٹل ییٚمہِ ؤرۍیہِ زیوٗنمُت اوس ۔
یہِ کتھ چھےٚ ونٕنۍ لایَق زِ ییٚمہِ ؤرۍ یہِ روٗد ثانیہ مرزایہِ ہُنٛد عروٗج زوال پزیٖر تہٕ ۂژِس لَگنہٕ کِنۍ روٗز ثانیہ سرِ دَس ہَتَن کھلاڑٮ۪ن ہٕنٛدِ لِسٹہٕ نٮ۪بر تہِ ۔
وۄنۍ کرَو أسۍ کرکٹ میٚچہٕ کِس مشہوٗر کھلٲڑۍ تہٕ کپتان دھونی ہٕنٛز کتھ ۔
دھونی سٕنٛدِ ناوٕ منسوٗب مٔنٛدر بناونَس پٮ۪ٹھ ٹھاک رانچی منٛز چُھ بھارتی کرکٹ ٹیٖمٕکۍ کپتان مہیندر سِنٛگھ دھونی سٕنٛدِ ناوٕ سۭتۍ منسوٗب مٔنٛدر تعمیٖر کرنَس پٮ۪ٹھ تنازعہٕ کھڑا گوٚمُت ۔
دھونی فین کلبَن چُھ تٔمۍ سٕنٛدِس آبٲی قصبہٕ رانچی ناوَس سۭتۍ منسوٗب مندر تعمیٖر کرنُک منصوبہٕ بنومُت ۔
مندر تعمیٖر کرنُک منصوبہٕ بنومُت تہٕ اَتھ پٮ۪ٹھ ٲس گۄڈٕ کٲم شروٗع سَپٕزمٕژ مگر دھونی سٕنٛدٮ۪و گرِ کٮ۪و ، دوستو تہٕ بیٚیَو نفرَو چُھ مندر تعمیٖر کرنَس پٮ۪ٹھ اعتراض کوٚرمُت ۔
رسایَو ہُنٛد وَنُن چُھ زِ ہیٚنٛدۍ مزہَب مُطٲبِق ہیٚکہِ نہٕ کٲنٛسہِ زنٛدٕ شخصہٕ سٕنٛدِس ناوَس پٮ۪ٹھ مندر بٔنِتھ ۔
دھونی فین کٕلَب کٮ۪ن رُکنن ہُنٛد مانُن چُھ زِ بھارتَس منٛز چُھ کرکٹس مزہبُک دَرجہٕ حٲصِل تہٕ دھونی چُھ اَمیُک خداے ییٚمہِ کِنۍ تِہُنٛدِ ناوٕ سۭتۍ منسوٗب مندر ضروٗر بنَن گَژھ ۔
اَمہِ پَتہٕ ترٛاوَو أسۍ اَکھ نَظَر أکِس نٔوِس پہلوٗوَس کُن یَتھ منٛز أسۍ کتھ کرَو دۄن تِتھین کھلٲڑِیَن ہٕنٛز یِم پٔزۍپٲٹھۍ چھِ یِوان ماننہٕ کرکٹ تہٕ مُکہٕ_بٲزی ہٕنٛدۍ پرٛزَلوٕنۍ تارَکھ ۔
کرکٹ تہٕ مُکہٕ_بٲزی ہٕنٛدۍ پرٛزلہٕ_وٕنۍ تارک ٹیِنٛڈولکر تہٕ ٹایسن سٔچِن ٹنڈولکر اَکھ بی_مِثال بیٹٕس_مین سٔچِن ٹنڑولکر چُھ ہِنٛدوستانُک نامور کرکٹ کھلٲڑۍ دُنۍیِہُک اَکھ بےِمِثال بیٹٕس_مین ییٚمۍ پننہِ مُنٛفرد اَنٛدازٕ سۭتۍ کرکٹ کِس دُنۍیاہَس منٛز واریاہ حٲرت انٛگیز کارنامہٕ اَنٛجام دِتۍ ۔
مَہرازَس چُھ یِوان پوٗر مَشرِقَس کُن بُتھ کرنہٕ ۔
اَمہِ پَتہٕ چھےٚ یِوان یٔزمَن باے تہٕ چھےٚ مَہرازَس آلَتھ کڑان تہٕ رٔتنہٕ ژٲنٛگِچ آلوان ۔
اَمہِ پَتہٕ چُھ یِوان مہرازَس نابَد رَژھا یا مِٹھایہِ رَژھا کھِیاونہٕ ۔
اَمہِ ساتہٕ چھےٚ وَنوَن واجنہِ کینٛہہ ہُرۍ وَنوان آسان ۔
اَمی وِزِ چُھ مہرازٕ سٕنٛدۍ یار دوس مہرازَس تہٕ انٛگَن ٲشناوَن فوٹوٗ تُلان ۔
یہِ چھےٚ اَکھ یادگاراہ روزان ۔
وۄنۍ چھِ بَٹہٕ وَنوَن واجنٮ۪ن علاوٕ تہٕ بَٹہٕ چالہِ وَنوَنہٕ علاوٕ مُسلمان چالہِ تہِ وَنوٕناوان ۔
یہِ چُھ پوٗرٕپٲٹھۍ مُسلمان چالہِ یِوان وَنوٕناونہٕ , کھڈا کھڈا روٗزِتھ تہٕ تَھدِہَٹہِ ۔
یِمہٕ وَنوَن واجنہِ چھےٚ مہرازَس توٚتام سۭتۍ نیران ، یوٚتام مہرازٕ گاڈِ یا کارِ کَھسان چُھ ۔
اَمہِ پَتہٕ چھےٚ تِمہٕ واپَس یِوان تہٕ چھےٚ اَتھ ویٖگِس پٮ۪ٹھ نَژان تہٕ اَتھ منٛز چھےٚ یٔزمَن باے تہِ نَژان تہٕ وَنوان ۔
اَتھ کھٮ۪نَس منٛز چھےٚ اَکھ کتھ یاد تھاوٕنۍ لایَق زِ گۄڈٕ چُھ یِوان بَرفی ٹُکرٕ مہرازَس کُن پِلناونہٕ یہِ چُھ اَتھ أڈ رَژھا ژَٹان ۔
پَتہٕ چُھ یوٚہَے ژھیوٚٹ بَرفی ٹُکرٕ مہریٚنہِ یِوان دِنہٕ ۔
یہِ تہِ چھےٚ اَتھ أڈ رَژھا ژٹان تہٕ تَو_پَتہٕ چھےٚ یٔہَے ژھیٚٹ بَرفی یِوان مَہرازَس بیٚیہِ آپراونہٕ ۔
یہِ چُھ اَتھ رَژھا ژٹان تہٕ تَوپَتہٕ چھےٚ یہِ ٲخرَس پٮ۪ٹھ یِوان بیٚیہِ مَہریٚنہِ دِنہٕ ۔
یہِ سَپدِتھ چھےٚ یٔزمَن باے مہریٚنہِ ہٕنٛز مَنَن مال کھولان تہٕ مَہرازَس گنٛڑان تہٕ مَہرازٕ سٕنٛز مَنَن مال کھولان تہٕ مَہریٚنہِ گَنٛڈان ۔
یہِ چُھ اَکہِ قٔسمٕچ سیِہرٕ چالہِ تِلہٕ_دوٚد ہِیُو آسان یُس مَہرازَس تہٕ مَہریٚنہِ چُھ یِوان پانہٕ وٲنۍ بَدلاونہٕ ۔
یٔزمَن بایہِ ہٕنٛز کٲم چھےٚ ییٖژٕے آسان ۔
اَمہِ پَتہٕ چھِ مَہرازٕ سٕنٛدۍ یار دوس کیمراہَن ٹِسراے تُلان ۔
تہٕ یِم چھِ مَہرازَس تہٕ مَہریٚنہِ اَتٮ۪تھ فوٹوٗ تُلان ۔
یِم چھِ مَہرازٕ سٕنٛدِ زِنٛدگی ہٕنٛدۍ یادگٲری فوٹوٗ آسان ۔
اَتہِ چُھ بُتھِس آبہٕ پَھشاہ دِتھ بھگوانَس آبہٕ گڈا ترٛاوان تہٕ رٔتٕنٛدیٖپھ زالان ۔
اَوٕ پَتہٕ چُھ سُہ مٔندر برٛہمنَس دٔکھنا دِتھ واپَس تٔتھۍ جایہِ واتان ییٚتہِ سالَر أمِس پرٛاران چھِ آسان ۔
مہرازٕ چُھ وۄنۍ سالرَن سۭتۍ جوٚلوٗسہٕ کِس صوٗرتَس منٛز کۄرِ والٮ۪ن ہُنٛد گَرٕ اَژان ۔
سۄاگَتھ کوچہِ گٔلِس پٮ۪ٹھ وٲتِتھ چُھ کۄرِ مول تہٕ گوٚبرٕ مول پانہٕ_وٲنۍ نالہٕ_موٚت کران ۔
کۄرِ مول چُھ گوٚبرٕ مٲلِس ، یُس تَس وۄنۍ سوٚنۍ چُھ آسان ، پوشہِ مالاہ یا ہاراہ نٲلۍ ترٛاوان ۔
اَمہِ پَتہٕ چُھ یِوان مَہرازَس ہار ترٛاونہٕ ۔
کۄرِ مٲلۍ سٕنٛدۍ سٲری انٛگ ٲشناو چھِ آسان کوچَس دۄیَو طَرفَو لٲنہِ ۔
تہٕ یِتھۍ کٔنۍ چُھ یِوان مَہرازَس تہٕ سالرن سواگَتھ کرنہٕ ۔
اَتھ جوٚلوٗسَس منٛز چُھ مہرازٕ آسان رازٕ ہِیُو وارٕوارٕ پٔرۍ ترٛاوان ترٛاوان پَکان ۔
تہٕ یِتھۍ کٔنۍ چُھ یہِ جوٚلوٗس تَتھ جایہِ واتان ییٚتٮ۪تھ کۄرِ والٮ۪و مہرازَس تہٕ مَہریٚنہِ ہٕنٛدِ خٲطرٕ ویوٗگ چُھ آسان ترٛوومُت ۔
اَمہِ علاوٕ چھےٚ أتھۍ یِوان وۄکھنَے کرنہٕ تہٕ بیٚیہِ چھِ یِوان تِم ہُرۍ تہِ وَننہٕ یِم میکھلہِ پٮ۪ٹھ چھِ دَرٕج آمٕتۍ کرنہٕ ۔
أتھۍ سۭتۍ چھےٚ گوٚبرٕ وٲلۍ سٕنٛدۍِ گَرِ وۄسہٕ درٛۄسہٕ ہَنا کم پَہَن گژھان ۔
مہرازٕ چُھ آسان درٛامُت ۔
ییٚمِس یَتھ جایہِ موقعہٕ چُھ یِوان سُہ چُھ تٔتھۍ جایہِ لَر ترٛاوان تہٕ راتُک وٕنیٚنٛدُر دوٗر کرنٕچ کوٗشِش کران ۔
ٲخٕر تَس چُھ آسان شامَن نۄش یِنَس پٮ۪ٹھ تہِ کھڈا روزُن ۔
وۄنۍ پانَس وُنۍکٮ۪نَس آرام دِن نہٕ تہٕ اَدٕ کیاہ کرَن ۔
سارِوٕے برٛۄنٛہہ چُھ آسان مہرازٕ سٕنٛز جیٖپ کار یا گُر ۔
یہِ چُھ آسان جان پٲٹھۍ پوشو ، پوشہِ مالَو , تہٕ واوٕ مالَو سۭتۍ سجٲوِتھ مہرازٕ چُھ أتھۍ پٮ۪ٹھ سوار سَپدان ۔
أمِس سۭتۍ چُھ پوٚت مَہرازٕ ، شنٛکھ وول ، چٔھترٕ وول تہٕ بیٚیہِ کانٛہہ لٲزِمدار شُر آسان ۔
اَمہِ پِتہٕ چھِ باقٕے سالَر کارَن تہٕ گاڈِ بیترِ منٛز سَوار سَپدان ۔
سالَر کۭتِیاہ گَژھن سۭتۍ آسٕنۍ تہٕ تِمَن ہٕنٛدِ کھٮ۪نُک چٮ۪نُک تہٕ روزنُک کیاہ اِنتِطام آسہِ ۔
یِمہٕ ساریٚے کتھٕ چھےٚ اَتھ منٛز درٕج آسان ۔
اَتھ چھِ ‘ لٔگنہٕ چیٖرۍ ’’ ’ وَنان ۔
اَمہِ پَتہٕ چُھ یِوان أمِس نابَد نوصلاہ تہٕ گٮ۪و ژۄچہِ تٔنزا دِنہٕ ۔
بیٚیہِ چُھس یِوان مٔدریر سیراہ تہِ دِنہٕ ۔
یہِ چُھ لَگنہٕ چیٖرۍ پَرنہٕ پَتہٕ ، یہِ سامانہٕ ہَن تولِیایہِ گَنٛڈان یٔزمَن چُھس رۄپیہ اَکھ ہَتھ تہٕ اَکھ ، اَکھ ہَتھ اَکونزاہ یا پَننہِ مقدوٗرٕ موٗجوٗب ہَتھ جوراہ تارٕ دِوان تہٕ جان پٲٹھۍ رۄخصَت کران ۔
یِمَو ژۄچَو تہٕ نابدٕ منٛز چُھ یہِ ہَنا جاراہ مہریٚنہِ نٔوۍیٖدٕ رٔنٛگۍ دِوان تہٕ باقٕے چُھ واپَس گَرٕ نِوان ۔
کۄرِ مٲلۍ والٮ۪ن چُھ گوٚبرٕ مالٮ۪ن ہٕنٛز سٲرٕے شیٚچھۍ خَبَر وَنان ۔
لَگنہٕ چیٖرۍ ہٮ۪تھ ییٚمہِ ساتہٕ برٛہمَن چُھ گَرٕ اَژان ، اَمہِ وِزِ چُھ وَنوُن آسان ۔
/ ۲ / سیِہرٕ گَنٛڈُن : مَہرازٕ ییٚلہِ پلووَلَو لٲگِتھ چُھ تَیار گَژھان تَس چُھ یِوان دَستار گنٛڈنہٕ تہٕ یِتھۍ کٔنۍ چُھ سُہ ییٚنہِ وٲلِس سۭتۍ نیرنہٕ خٲطرٕ تیار سَپدان ۔
ویوٗگ / کۄرِمٲلۍسٕنٛدِگَرِ / : مہرازٕ چُھ ویٖگِس کَھسان تہٕ چٔھترٕ وول چُھ چٔھتٕر ہٮ۪تھ پَتہٕ کنہِ روزان ـ شینٛکھ وول چُھ شینٛکَھس ناداہ ناداہ دِوان ۔
یہِ زَن گۄو کۄرِ والٮ۪ن حٮ۪س دیُن زِ مَہریٚنۍ وٲلیوٗن ویٖگَس پٮ۪ٹھ ۔
تانۍ چُھ کۄرِ ہُنٛد مام یِوان کۄرِ وٲلِس دُسہٕ منٛگنہِ ۔
اَمیُک مَطلَب چُھ زِ مَہریٚنۍ چھےٚ ویٖگِس پٮ۪ٹھ وَسنہٕ خٲطرٕ تَیار ۔
دُسہٕ ہٮ۪تھ چُھ مام نیران تہٕ چُھ مَہریٚنہِ ویٖگِس پٮ۪ٹھ والان ۔
تام چھےٚ مہرازٕ سٕنٛز ہَش مٲج یِوان تہٕ چھےٚ مَہرازَس تہٕ مہریٚنہِ یۄسہٕ مَہرازَس کھوٚوٕرۍ کِنۍ چھےٚ یِوان تَھونہٕ ، آلَتھ کڑان ۔
پَتہٕ چُھ یِمَن یِوان ‘‘ رٔتنہٕ ژٲنگۍچہِ ’’ سۭتۍ آلَتھ کڑنہٕ ۔
یِم چھِ آسان توٚملہٕ آٹِکۍ ٲمۍ ژٲنٛگۍ یِمَن منٛز تیٖلہٕ بۄڈنٲوِتھ سۄیَژِ چھےٚ آسان ۔
ژٲنٛگۍ چھِ یِوان زالنہٕ تہٕ زٲلِتھ چھےٚ یِوان دۄشوٕنۍ آلَتھ کڑنہٕ ۔
اَمہِ علاوٕ چھےٚ یٔزمَن باے یِمَن برفی رژھا رژھا کھیاوان ۔
یٔزمَن چُھ سارِنٕے بٮ۪ہناوان تہٕ سارِنٕے ہُنٛد خیال تھاوان ۔
یِتھٕۍ کٔنۍ چُھ ییٚنہِ وول روانہٕ سَپدان ۔
میٖلاہ پٔکتھ چُھ یٔزمَن سُنٛد کانٛہہ کارکُناہ وۄتھان تہٕ سالرٮ۪ن سِگریٹ بیترِ بٲگران ۔
یہِ سِلسِلہٕ چُھ توٚتام جٲری روزان یوٚتام بَراتھ کۄرِمٲلۍ سُنٛد گَرٕ واتان چھےٚ ۔
واپَس یِوان یِوان چھےٚ وَنوَن واجن بیٚیہِ وَنوان ۔
یِتھ کٔنۍ چُھ ییٚنہِ وول کۄرِ مٲلۍ سٕنٛدِس علاقَس منٛز واتان ۔
کارِ منٛزٕ وَسوُنُے چُھ شنٛکھ وول شنٛکھٕ شَبداہ کران یُتھ زَن کۄرِ والٮ۪ن خَبَر گَژِھ زِ ییٚنہِ وول ووت ۔
مَہرازٕ چُھ کارِ منٛز بۄن وَسان ۔
پَلوَن گیٚنۍ کڑان تہٕ چٔھترٕ وول چُھ مَہرازَس پَتہٕ کُنہِ چٔھترٕ ہٮ۪تھ کَھڈا روزان ۔
مَہرازٕ چُھ اَمہِ پَتہٕ منٛدرس پٮ۪ٹھ گژھان ۔
سجاو سِنٛگار چُھ طَیَس پٮ۪ٹھ آسان ۔
سۄنہٕ مہریٚنہِ ہِشہِ چھےٚ پَنُن پَنُن سَکھراو کران ، یِتھۍ کٔنۍ چھِ دۄنوَے دٔرۍ کَلہٕ_کاپہِ سان اَتھ کامہِ نکھٕ والنٕچ کوٗشِش کران ۔
لٔگٕن لٔگٕن چُھ مَہرازٕ سٕنٛدِ یِنہٕ پٮ۪ٹَھے شروٗع گژھان ۔
اِسلٔے کرو أسۍ گوٚبرٕ مٲلۍ سٕنٛدِ گَرٕ نیرنہٕ پٮ۪ٹَھے شروٗع ۔
گوٚبرٕ مٲلۍ سُنٛد ییٚنہِ_وول سَکھراونَس سۭتۍ چھِ یِم رسٕم شٲمِل : ـ / ۱ / ـ لٔگنہٕ چیٖرۍ ـ / ۲ / ـ سیٚہرٕن گنٛڈٛن ـ / ۳ / ـ ویوٗگ ـ / ۴ / ـ ییٚنہِ_وول ـ / ۱ / ـ لٔگنہٕ چیٖرۍ : ییٚنہِ والہِ کالہٕ چُھ کۄرِ وٲلۍ سُنٛد کۄلہٕ گور گوٚبرٕ وٲلۍ سُنٛد گَرٕ یِوان ۔
دَروازَس پٮ۪ٹھ واتوُنُے چھےٚ تَس یِوان , آلَتھ کڑنہٕ , توپَتہٕ چُھ یِوان یَزتہٕ سان أکِس کُٹِھس منٛز کھارنہٕ ۔
أمِس چُھ یِوان دۄداہ قہواہ دِنہٕ تہٕ اَتھ سۭتۍ چَھس یِوان جان ژۄچہِ ہَنا یا مِٹھایہِ ہَنا دِنہٕ ۔
اَمہِ پَتہٕ چُھ سُہ اَکھ چھٲپۍ رۄکاہ ہِیُو چنٛدٕ منٛزٕ کڑان ۔
یہِ چُھ آسان دیوناگری رسمِ خطَس منٛز لیٚکھِتھ ، ، تہٕ شیٚچھۍ خَبَر لیٚکھنٕچ جاے چھےٚ اَتھ خٲلی آسان ۔
اَتھ منٛز چُھ برٛۄنٛہٕے مُقرر کرنہٕ آمتِس دۄہَس پٮ۪ٹھ کۄرِ وٲلۍ سٕنٛدِ طرفہٕ گوٚبرٕ وٲلِس ییٚنہِ_وول ہٮ۪تھ یِنَس دَاوَتھ دِنہٕ آمٕژ ۔
اَمی وِزِ چُھ تَس سیِہرٕ گَنٛڈنہٕ یِوان ۔
یہِ چُھ آسان تِلہٕ_دار اَکھ جالاہ ہِیُو یُس تَس دَستارَس پٮ۪ٹھ چُھ یِوان گنٛڈنہٕ ۔
یہِ چُھ تَس سٲرۍسٕے بُتِھس پٮ۪ٹھ پٮ۪وان تہٕ اَکہِ قٔسمٕچ زیبٲیی ہِش دِوان ۔
اَمہِ ساتہٕ چُھ تَس اِسبَنٛد بیترِ یِوان زالنہٕ تہٕ ڈٮ۪کَس میٖٹھۍ دِنہٕ یِوان ۔
اَمہِ پَتہٕ چھےٚ أمِس یِوان پوشہِ مالہٕ ، نوٹہٕ مالہٕ تہٕ خٲلی مالہٕ ترٛاونہٕ ۔
مہرازٕ چُھ وۄنۍ ویٖگِس پٮ۪ٹھ نیرنہٕ خٲطرٕ تَیار آسان ۔
بَراتھ چھِ سٲرٕے آنٛگنَس منٛز شٛنکھ شَبدا دِوان تہٕ یِتھۍ کٔنۍ مہرازَس ژٮ۪تَس پاوان زِ بَراتھ چھےٚ بۄنہٕ کنہِ اِنتِطارَس ۔
ـ / ۳ / ـ ویوٗگ / گوِبرٕوٲلۍسٕندِگَرِ / : آنٛگنَس منٛز چُھ مَہرازٕ سٕنٛدِ خٲطرٕ اَکھ گول تہٕ سوٚم جایا یِوان لِونہٕ اَتھ پٮ۪ٹھ چھِ رنٛگ_دار اَرکوٚشہٕ سۭتۍ یِوان اَکھ نقشہٕ_نِگٲری_دار گولا بناونہٕ ۔
مہرازٕ چُھ وَسان تہٕ ویٖگَس کَھسان ۔
أمِس چُھ اَکہِ طرفہٕ یِوان توٚملہٕ ٹوکاہ تھاونہٕ اَتھ پٮ۪ٹھ چُھ یٔزمَن رۄپیاہ تہٕ نوٗنہٕ ہَنا تھاوان ۔

حٔقیٖقت چھےٚ یہِ زِ یہِ لَفٕظ چُھ سنسکرت رسنہٕ نِش درٛامُت ۔
سنسکٕرٛتَس منٛز چھِ رَسنہٕ وَنان زیٚوِ ۔
تٔمۍ ووٚن نہٕ تِمَن کہیٖں ۔
تٔمِس توٚر فِکرِ زِ یِہ سٲرٕے شیطٲنی محمدٕنۍ ۔
اَمہِ وِزِ کوٚر تٔمۍ بَرداش ۔
تٔمۍ زون اگر بہٕ یِمَن جواب دِمہٕ یِم تَراوَن کارخانہٕ أتۍتھٕے تہٕ ژَلَن ۔
شوٹَھس گژھِ سٮ۪ٹھاہ نۄقصان مگر أنٛدرٕۍ کنۍ روٗدُس یوٚہے کیُل ہِیُوٖ ۔
دوٚیمہِ دۄہہِ کٔر تٔمۍ کٲم ۔
عبداللہ تہٕ غنی أنِن خلَوکھ پٲٹھۍ گرٕ ـ تَتہِ سمجھٲوِن ۔
ماز بَتہٕ کھیٲوِن ، تِمَن ووٚنُن بہٕ کھالوٕ تۄہہِ تنخواہَس رۄپَے رۄپَے ۔
وُچھوَے تہٕ اِنسان چُھ کَنہِ کھۄتہٕ سَخ مانَوے تہٕ پوشہِ کھۄتہٕ ٲویُل ۔
مےٚ تہِ اوس اکہِ وقتہٕ پَنُن پان شٔسترُو باسان ۔
مےٚ چَھ نہٕ خَبَر زِ پَزہا ژے پَننہِ دِلٕچ کَتھ وَنٕنۍ ، خاص_کٔرِتھ تَمہِ ساعتہٕ ییٚلہِ مےٚ گُمنام ، ازوٗن تہٕ کٔھٹِتھ پَزِ روزُن ۔
بَنان چُھ زِ یِتھٕ کٔنۍ ہٮ۪کہٕ بہٕ چانٮ۪ن نَظرَن منٛز دۄگنہٕ گۄناہ_گار بٔنِتھ ۔
تٔمِس اِنسان سٕنٛد پٲٹھۍ یُس زیبرا کراسِنٛگ پٮ۪ٹھ شرٛابَس منٛز مَس ٲسِتھ تٔمِس شُرِس گٲڈۍ پیٚٹھۍ چُھ کھالان یَس کٕہیں سۭتۍ آسہِ تہٕ پَتہٕ چُھ وٲرۍدات چہِ جایہِ پٮ۪ٹھ ژَلان ۔
مگر ژے چُھے مسیٖحَن اِنسٲنی دۄکَھن تہٕ دادٮ۪ن ہُنٛد مُقدَس اِدراک عطا کوٚرمُت تہٕ یہِ کری ژے پَزی میٲنِس شٔدیٖد روٗحٲنی کربَس مَنٛز شٔریٖک سَپدٕنَس مَنٛز مدد ۔
﻿أکس اِنسانَس چُھ پرٮ۪تھ وِزِ پانَس مَنٛز تِتھۍ آداب پٲدٕ کرٕنۍ آسان یِم تٔمِس پوٗرٕ_پٲٹھۍ اَکھ مُکمَل اِنسان بناوَن تہٕ سماجَس مَنٛز روزنہٕ خٲطرٕ تٔمۍ سٕنٛدِ خٲطرٕ اَکھ مُستقِل جاے بناوَن ۔
تِکیازِ یَتھ سماجَس مَنٛز چِھ پرٮ۪تھ أکس اِنسانَس أمِہ کٮ۪ن طریٖقَن تہٕ او٘صوٗلَن پٮ۪ٹھ پَکن پٮ۪وان ۔
تہٕ پرٮ۪تھ اَکھ سماج چُھ پرٮ۪تھ أکس اِنسانَس تی٘لی اَپناوان یی٘لہِ تٔمِس مَنٛز تِتھۍ طریٖقہٕ تہٕ آداب آسَن یِم أکس تہزیٖب_یافتہٕ شہریٖیَس مَنٛز گَژھن آسٕنۍ ۔
یِمَن طریٖقَن تہٕ آدابَن مَنٛز چُھ یِوان اِنسان سُنٛد پرٮ۪تھ_کانٛہہ پہلوٗ ، تٔمۍ سُنٛد وۄتُھن بِہُن ، نیرُن اَژُن ، کتھ کرٕنۍ تہٕ پوشاکھ لاگنٕکۍ آداب ۔
سارِنٕے برٛۄنٛہہ کرَو أسۍ کتھ پوشاکھ لاگنٕک آدابَن ہٕنٛز ۔
سانٮ۪و بُزَرگَو ، بَڑٮ۪و تہٕ سانٮ۪و پیغمبَرَو تہِ چُھ فَرمومُتھ پوشاکھ لاگنٕکٮ۪ن آدابَن مُطلِق ۔
سۭتۍ سۭتۍ چُھ یی٘مہِ کتھہِ ہُنٛد تہِ حجس تھاوُن زِ پۄشاکھ گژھہِ موسمی اثراتو نِشہِ تہٕ جسمس حفاظت کرُن تہٕ حسن و جمالُک مقصد تہِ پوٗرٕ کرُن ۔
پۄشاکھ گژھِ نہٕ تیُتھ تہِ لاگُن یی٘مہِ سۭتۍ انسان تماشہِ ہیوٗ باسہِ تہٕ لوٗکن میلہِ اَسہٕنُک موقہٕ ۔
پوشاکھ ووت نہٕ یِتَھے_پٲٹھۍ اَسہِ تام ، یہِ چھی٘ہ پٔزۍ_پٲٹھۍ اللہ_تعالٰی سٕنٛزٕے مہربٲنی یی٘مۍ اَسہِ تام پوشاکھ واتہٕ_نوو ۔
پۄشاکھ لاگنہٕ وَقتہٕ سوٗنچِو یہِ_زِ یہِ چُھ اﷲ_تعالٰی سٕنز نعمت ، یۄسہٕ اﷲ_تعالٰی ہَن صِرف انسانس چُھ عطا کٔرمٕژ ۔
باقی مخلوٗقات چھِ اَمہِ نِشہِ محروم ۔
یہِ ہٮ۪کو أسۍ یِتھہٕ_پٲٹھۍ تہِ ؤنِتھ زِ اللہ_تعالٰی سٕنٛز باقٕے مخلوٗق یَتھ مَنٛز جانور بیترِ چِھ یِوان ، تِمۍ چِھ پورٕ_پٲٹھۍ اَمہِ نعمتہٕ نِش محروٗم ۔
مَگَر پوشاکھ تہِ چُھ سُہ لاگُن یُس أکس اِنسان سٕنٛز شرم و حیا پرٮ۪تھ وَقتہٕ پننٮ۪ن حدن مَنٛز تھاوِ ۔
بہترین پۄشاکھ چُھ تقواہُک پۄشاکھ ۔
تقواکہِ پۄشاککھ مطلب چُھ بٲطنی پٲکیزگی تہِ تہٕ ظٲہری پرہیز_گٲری ہُنٛد پۄشاکھ تہِ ۔
یعنی تیُتھ پۄشاکھ لٲگِو یُس شٔرۍیت_چن نظرن منٛز پرہیز_گارَن ہُنٛد پۄشاکھ آسہِ ۔
تہٕ بہٕ چُھس پنُن پان چٲنِس حةفاظتس منٛز دِوان اَمہِ پۄشاکھچہِ بُرٲیی نِشہِ تہٕ تمہِ مقصدٕ خٲطرٕ یی٘مہِ خٲطرٕ یہِ بناونہٕ چُھ آمُت ’ ۔
اَمہِ دُعاہُک مطلب چُھ یہِ_زِ ‘‘ خۄدایا م٘ے دِ توفیق زِ بہٕ کرٕ چون بخشمُت پۄشاکھ تِمنٕے مقصدن خٲطرٕ استمال یِم ژ٘ے نزدیک پٲکیزٕ مقصد چِھِ ۔
مے٘ دِ توفیق زِ بہٕ کرٕ اَمہِ سۭتۍ پنٕنۍ ستٕر_پوٗشی تہٕ بے_شرمی ، بے_حیٲیی ہٕنٛز وکتھو نِشہِ تھاوٕ پنُن ظٲہر تہٕ بٲطن پاک تہٕ شرۍییت کةن حو٘دوٗدن منٛز روٗزِتھ کرٕ بہٕ اَمہِ سۭتۍ پننہِ جسمٕچ حفاظت تہٕ یہِ بناون بہٕ پننہِ خوبصوٗرتی ، زینت تہٕ جمالُک زٔرِیہِ ۔
پۄشاکھ لٲگِتھ کر نہٕ بہٕ پننہِ بدنٕچ نمٲیش تہٕ اظہار نہ کرٕ غروٗر تہٕ تکبُر نہ پُھٹراوٕ بہٕ یہِ نعمت استمال کرنہٕ وِزِ شرۍیتٕکۍ حو٘دوٗد یِم ژ٘ے پننةن بندن پةٹھ مقرر چھہِ کٔرِمٕتۍ ۔
اَمہِ پَتہٕ کرَو تِمَن سارِنٕے چیٖزَن مُطلق کتھ یِم اِنسانَس خٲطرٕ ہَر وِزِ ضو٘روٗری چِھ آسان ۔
تٔمۍ سٕنٛز کتھ کرٕنۍ یعنے تٔمۍ سُنٛد طریٖقہٕ چُھ پریتھ وِزِ تٔمۍ سٕنٛدین کتَھن مَنٛز ظٲہِر سپدان ۔
یُتھ تٔمۍ سٕنٛدِ کتھ کرنُک طریٖقہٕ آسہِ تیُتُھے چُھ سُہ پانہٕ تہِ آسان یعنے اَگر سُہ نرمی سان کتھ آسہِ کران تی٘لہِ چِھ تٔمِس مَنٛز تہِ نرمی آسان ، تہٕ اگر سُہ تیزرٕ سان کتھ آسہِ کران یا تٔمۍ سُنٛد کتھ کرنُک لحجہِ آسہِ تیز پَہن سُہ چیٖز چُھ یی ظٲہِر کران زِ سُہ تہِ چُھ تیزٕے پَہَن ۔
بیاکھ اَکھ چیٖز گوو تٔمۍ سُنٛد دُنۍیِہ کین لوٗکن سۭتۍ وۄتھنہِ بی٘ہنُک طریٖقہٕ ۔
دُنۍیِہُک لوہ_لَنٛگر پَکناوُن ، پَنٕنۍ شُرۍ بٲژ پالٕنۍ چُھ اِنسان سُنٛد سارِوٕے کھۄتہٕ بو٘ڑ فرٕض تہٕ یہِ انجام دِنہٕ خٲطرٕ چُھ پریتھ أکس اِنسانَس ہَر مُمکن کوٗشِش آسان کرٕنۍ زِ تٔمۍ سٕنٛزِ زِندٕگی سۭتۍ جِڑِتھ ہَر_کنہِ شوبَس مَنٛز روزِ پنُن اکھ توازُن بَرقرار ۔
تہٕ یُتھ نہٕ اَکھ چیٖز واتہٕ_ناوِ بی٘یِس چیٖزَس کنہِ تہِ طریٖقہٕ کانٛہہ تہِ زریر ۔
مگر تِتُھے اوس سُہ اَسٕنٛدِ کھۄتہٕ نَل نکھٕ کِنۍ زیادٕ اوٚر دوٚر باسان ۔
یہِ اوس پَزر نتہٕ ٲسۍ دۄنوَے اَکی ذٲژ ہٕنٛدۍ تہٕ ژور آنہٕ کم یا ژور آنہٕ زیادٕ اَکی مۄلہٕ تہِ ۔
دۄہ گٔژھِتھ شامَس ییٚلہِ سٲری شوٚنٛگۍ ۔
شاہ مال ٲس بِسترَس منٛز شُرِس لَرِ سٲوِتھ دۄد چاوان ۔
اَمہِ اوس شانٛدٕ گوٚنٛدِس پٮ۪ٹھ کھۄنہٕ وَٹھ ٹھیٖکہٕ رٲوِتھ اَتَھس پٮ۪ٹھ کلُک سورُے بار روٚٹمُت ۔
یِتھ کٔنۍ ٲس یہِ شُرِس دۄد چاوان مگر أنٛدٕرۍ أنٛدرۍ ٲس یہِ أمۍسٕے کۄکرٕ سُنٛد سونٛچان ۔
“ کۄکُر وُچھِتھ کیاہ آسٮ۪س بِژِ وٲنِجہِ نوٗن پیوٚمُت ۔
مےٚ ہَے وُنہِ تہِ اَہنٛز سۄ کَتھ اوم تیٖل زَن چھےٚ تیلان ۔
وۄنۍ گژھِ اَسُنٛد کۄکُر یور ترُن ۔
زنٛگہِ نِشہِ کوٚرُم تیٚلہِ چَھسَے نہٕ اَدٕ ۔
بہٕ اوسُس نہٕ کۄمَن کۄمَن خٲطر اَنان ۔
مگر بازے چُھ اِنسانَس تٔمۍ سٕنٛدٮ۪ن جَزبَن تہٕ خیالَن ہُنٛد لحازا کرُن پٮ۪وان ، یُس تَس سارِوٕے کھۄتہٕ زیادٕ ٹوٹھ آسہِ اَدٕ تِم جزبہٕ پانَس وٮ۪ژن یا وٮ۪ژن ۔
یہِ وَنُن نَے منٛدچھ گژھِ مےٚ ٲس شانٛتہٕ پَنہِ زُوٕ کھۄتہٕ ٹٲٹھ ، تَوے اوسُس بہٕ تَسٕنٛد سٲری ناز تُلان ۔
نَتہٕ مےٚ ٲس نا خَبَر زِ دٲدِس دَگہِ گژھِ علاج واتُن دوا واتُن ۔
ییٚمہِ لولہٕ تہٕ شوقہٕ میٲنۍ آشیٚنۍ سنٛکرٲژ دۄہَہ فاقہٕ ٲس تَھوان سیتہ نرایُن سٕنٛز پوٗزا کران گورو جِیس دیکھشا دِوان ۔
سٲرۍ سٕے مَحلَس پوٗرۍ بٲگران ۔
تَمہِ کِنۍ گوٚژھ نہٕ میانہِ زنانہِ کھورس کوٚنڑ اَژُن ۔
یُتُھے تَمہِ اَسمٲنۍ رنٛگہٕ عطر مٔتھِتھ سُہ لِفافہٕ ژَٹنہٕ خٲطرٕ کاغز تراش اَتَھس منٛز تُل یَتھ کھوٚورِس کوٗنَس پٮ۪ٹھ غنٛچدار شنگرٔنۍ گۄلاب نَصٕب کٔرِتھ اوس تٔمۍ سٕنٛدِ سوٚترِ ڈٮ۪کہٕ چَٮ۪ن کینٛژن گیٚنٮ۪ن مَنٛز وۄتلیوٚو کرب_ناکھ اِشتیاق ۔
گۄڈنِچٮ۪ن سٕطرَن منٛز تَراوُنَے وۄشلیوٚو تٔمۍ سُنٛد بُتھۍ شٔدیٖد مَلالہٕ سۭتۍ ۔
‘‘ یہِ کژھ شیطٲنۍ بَدتمیٖزی ’’ تَمہِ ووٚن پانَس سۭتۍ ۔
یہِ ہَبا گۄو سٮ۪ٹھاہ ناکارٕ “ پیڑرو سُنٛد دِل اوس برٛونٛٹھے وَنان زِ تَس چُھ کیاہتام وُلٹہٕ وُچُھن گورَن ؤنۍ نہٕ کانٛہہ نٔو کَتھ , مگر حالَس اوس سوال زِ ہُم نَفَر ہَے گٔیہِ تہٕ کوٚت گٔیہِ ؟
دَبٕے اگر لوٚگمُت آسٮ۪کھ ، توتہِ ما گژھٕہن وُڈِتھ ۔
خَبَر کیاہ دٔلیٖل چَھ ۔
گور اوس زورٕ زورٕ پَران ۔
میانٮ۪ن اَپرادن کرتہٕ شھٮ۪ما ـ ہی شیوٕ سمبھو ـ ہی مہادیو شمبھو ۔

ٹرانسپورٹ محکمہٕ کٮ۪ن ملٲزمن ہُنٛد پٔتۍمٮ۪و پانٛژو رٮ۪تو پٮ۪ٹھٕ لَگاتار احتجاج تہٕ ہَرتال کرنہٕ , سَڑکَن پٮ۪ٹھ پُلسہٕ دٔسۍ کالہٕ پَگاہ یِہُنٛد مار کھیوِن شُرۍ بٲژ ہیتھ سَڑکَن پٮ۪ٹھ نیرُن تہٕ سکوٗل بیگ زالٕنۍ چھِ تِم کَتھٕ یِمَو سۭتۍ اَکھ سوال چُھ ذہنَس منٛز وۄتلان ۔
سُہ گۄو یہِ زِ کیاہ جٲیز مُطالبہٕ پوٗرٕ کرناونہٕ باپَتھ چَھا وۄنۍ ضوِروٗری بنیوِمُت زِ گنٛڈ گَژِھ سرکٲرۍ ڈنٛڈٕ بَردارو دٔسۍ بےِوایہِ مار کھیوِن یا شُرۍ بٲژ ہٮ۪تھ سَڑکَن پٮ۪ٹھ نیرُن تہٕ اَدٕ واتہِ حٲکمَن تام آواز ؟
سونٛچنَس لایَق چھےِ کَتھ زِ پانٛژَن ریتَن پیِیہِ لَتَن ہُنٛد ماز وتَن لارنٲوِتھ یِتھٕ کٔنۍ ٹرانسپورٹ مٕلٲزِم سرکارَن کَتھ باتھ کرنہٕ باپَتھ ناد دِتھ أنۍ ، تِتھٕ کٔنۍ ہٮ۪کہٕ ہَن تِم یانے حٲکم واریاہ کال برٛۄنٛہہ تہِ کٔرِتھ ۔
شاید پِہَن نہٕ ٹرانسپورٹ مٕلٲزمن پَنہٕ نٮ۪ن شُرٮ۪ن ہٕنٛدۍ بیگ تہٕ تِہٕنٛزٕ کِتابہٕ زالنہِ ۔
بَچِہ مَنٛگتہٕ اَنُن چُھ سٮ۪ٹھاہ پرٛون رَسٕم ییٚمیُک حوالہٕ اَسِہ رامایِنہٕ منٛز تہٕ مَہابھارتہٕ منٛز تہِ میلان چُھ ۔
وۄں گۄو مَنٛگتہٕ اَننہٕ وِزِ چھِ بَٹَن منٛز کۄل تہٕ گُتھٕر ضوٚروٗر وُچھان ۔
جان کۄلُک تہٕ سۄگتھرِیُک پۄتُھر چھِ بَڑِ چاوٕ سان تہٕ طوالٔژ بغٲر یِوان اَننہٕ ۔
مُسلمانَن منٛز تہِ چُھ کانٛہہ کانٛہہ وُچھان زِ بَچہِ چُھ جان گَرُک کِنہٕ NULL پَتہٕ چھِ تٔسۍ بَچَس مَنٛگتہٕ اَنان یُس تِمَن باسان چُھ اَصلہِ ، گرُک ، مگَر اَکثَر لوٗکھ وٕچھان یِم چیٖز ۔
اگر کٲنٛسہِ موٗلہٕ پٮ۪ٹھٕ آسہِ نیٚچُو یا کوٗر زامٕژ , تَس چُھ پٮ۪وان سُہ ضوٚروٗر مَنٛگتہٕ دِیُن ۔
ہَرگاہ جوتشہِ موٗجوٗب یہِ پۄتُھر مَنٛگتہٕ دِنَس پنٕنۍ کینٛہہ ووٚجوٗہاتھ چھِ مگر سمٲجی طور چُھ یی باسان زِ سٲنۍ سماجَن چھیٚہ یہِ پۄتھرن پۄتھر پَراونٕچ اَکھ وَتھا تھٲومٕژ ۔
تہٕ یِتھٕ کٔنۍ نَہ چُھ لَگان ماجہِ مٲلِس دَنٛدَن تہٕ نَہ چھیٚہ مَنٛگتہٕ نِنہٕ وٲلِس ناوۄمیدی سَپدان ۔
أمِس مَنٛگتہٕ أنۍمٕتِس کۄرِ یا گوٚبرَس چھِ سٲری رَسٕم پَتہٕ أتھۍ مَنٛگتہٕ گَرَس منٛز پوٗرٕ یِوان کرنہٕ ۔
أمۍ سٕنٛد سٲری رَسٕم تہٕ رٮ۪واج چھِ یِوان أتھۍ مَنٛگتہٕ گَرَس منٛز کرنہٕ چاہَے تٔمِس زَرٕ آسَن کاسٕنۍ یا کَن آسَن ژۄمبٕنۍ یا خَتَن ہال آسٮ۪س کرٕنۍ ۔
یِم سٲری رَسٕم پوٗرٕ کرنہٕ سۭتۍ چُھ أمِس بَچَس تِہ باسان زِ یِمَے چھِ میٲنۍ اَصلی مول تہٕ موج ـ
